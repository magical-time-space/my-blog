        analyser.connect(audioCtx.destination)
    } catch (error) {
        console.log('~~~analyser:error', error)
    }
    //
    function destroyed() {
        try { audioCtx.close().then(function() {  }); } catch (error) { }

        }

    console.log('~~~analyser:dataArray', { dataArray, canvas });
    let [WIDTH, HEIGHT, bufferLength] = [canvas.width, canvas.height, analyser.frequencyBinCount];
    //
    function drawWaveline() {
        requestAnimationFrame(drawWaveline);
        analyser.getByteTimeDomainData(dataArray);
        canvasCtx.fillStyle = "#000000";
        canvasCtx.fillRect(0, 0, WIDTH, HEIGHT);
        canvasCtx.lineWidth = 2;
        canvasCtx.strokeStyle = "#32a35e";
        canvasCtx.beginPath();
        const sliceWidth = (WIDTH * 1.0) / bufferLength;
        let x = 0;
        for (let i = 0; i < bufferLength; i++) {
            const v = dataArray[i] / 128.0;
            const y = (v * HEIGHT) / 2;
            if (i === 0) {
                canvasCtx.moveTo(x, y);
            } else {
                canvasCtx.lineTo(x, y);
            }
            x += sliceWidth;
        }
        canvasCtx.lineTo(canvas.width, canvas.height / 2);
        canvasCtx.stroke();
    }
    //
    function drawRect() {
        requestAnimationFrame(drawRect);
        analyser.getByteFrequencyData(dataArray);
        canvasCtx.fillStyle = "#000000";
        canvasCtx.fillRect(0, 0, WIDTH, HEIGHT);
        var barWidth = (WIDTH / bufferLength) * 2.5;
        var barHeight;
        var x = 0;
        for (let i = 0; i < bufferLength; i++) {
            barHeight = dataArray[i] / 1;
            canvasCtx.fillStyle = '#410c0c';//'rgb(' + (barHeight+100) + ',50,50)';
            // canvasCtx.fillRect((-barHeight / 2), x, barHeight, barWidth);
            canvasCtx.fillRect(x,(HEIGHT-barHeight/2),barWidth,barHeight);
            x += barWidth + 1;
        }
    }
    //
    // Math.random()>0.45?drawRect():drawWaveline();
    //
    drawRect();
    return {analyser,source,audioCtx, destroyed};

    争与不争

    与人相争，赢的秘诀，就是审时度势，一定要远远超越的时机出现时，才能大张旗鼓的争，争的是终极版的赢，不是反复输输赢赢，搁置消耗，无终无极；所以。反之，不争。

    不争，则让。让不是被吞并、被掠夺、被侵占。是该割舍什么的思考。是做自我减法，减了肥，我还是我。
